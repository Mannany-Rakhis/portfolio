class QuizEngine:
  def __init__(self, bank):
    self.questions = self.shuffle(bank)
    self.score = 0
    self.penalty = -0.5 # Points négatifs

  def submit_answer(self, idx, user_idx):
    q = self.questions[idx]
    if user_idx == q.correct_idx:
      self.score += 1
    else: self.score += self.penalty
